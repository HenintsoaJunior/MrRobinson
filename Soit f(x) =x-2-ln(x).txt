Soit f(x) =x-2-ln(x)

1)Montrer qu'il existe une et seul solution de f(x)=0 dans l'intervalle [a;b] = [1;4]

2) Calculer lampda=b-a/(f(b)-f(a))

3) definit g(x,a,b)=x-lampda-f(x)

Montrer que f(x) =0 si et seulement si x=g(x,a,b)

4) Definier une fonction g en C

5) Ecrire une fonction en C

float ptfx(float x,float a,float b,float eps)

calculant la limite de la suite 
{x0=a+b/2  xn+1=g(xn,a,b)} en utilisant comme test d'arret |xn+1-xn| < epsilone


Utilise alors cette fonction pour resoudre f(x)=0 a eps=10^-6 

#include <stdio.h>
#include <math.h>

float f(float x) {
    return x - 2 - logf(x);
}

float g(float x, float a, float b) {
    float lambda = (b - a) / (f(b) - f(a));
    return x - lambda - f(x);
}

float ptfx(float x, float a, float b, float epsilon) {
    float xPrecedent, xCourant;
    xCourant = x;

    do {
        xPrecedent = xCourant;
        xCourant = g(xPrecedent, a, b);
    } while (fabsf(xCourant - xPrecedent) > epsilon);

    return xCourant;
}

int main() {
    // Exemple d'utilisation pour r√©soudre f(x) = 0 avec epsilon = 10^-6
    float a = 1.0;
    float b = 4.0;
    float suppositionInitiale = (a + b) / 2;
    float epsilon = 1e-6; // 10^-6

    float resultat = ptfx(suppositionInitiale, a, b, epsilon);

    printf("Une solution de f(x) = 0 est : %f\n", resultat);

    return 0;
}
